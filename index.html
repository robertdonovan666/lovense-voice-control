// ==================== Get AuthToken from Backend ==================== async function getAuthTokenFromBackend() { console.log('📍 Fetching authToken from backend...'); try { const response = await fetch(BACKEND_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' } }); speedSlider.disabled = false; cmdButtons.faster.disabled = false; cmdButtons.slower.disabled = false; sendPattern('fireworks'); } else if (cmd.includes('earthquake')) { state.currentPattern = 'earthquake'; } } function increaseSpeed() { if (state.currentPattern === 'earthquake') { showStatus('Speed control not available for Earthquake'); return; } if (state.currentSpeed < 100) { state.currentSpeed += 10; } else { playBeep(); showStatus('⚠️ Minimum speed reached'); } } // ==================== UI Event Listeners ==================== intensitySlider.addEventListener('input', (e) => { state.currentIntensity = parseInt(e.target.value);
